 46:     /**
 47:      * Replace all * wildcards in a URI by the current segment in that location
 48:      *
 49:      * @return  string
 50:      */
 51:     public static function segment_replace($url)
 52:     {
 53:         // get the path from the url
 54:         $parts = parse_url($url);
 55: 
 56:         // explode it in it's segments
 57:         $segments = explode('/', trim($parts['path'], '/'));
 58: 
 59:         // fetch any segments needed
 60:         $wildcards = 0;
 61:         foreach ($segments as $index => &$segment)
 62:         {
 63:             if (strpos($segment, '*') !== false)
 64:             {
 65:                 $wildcards++;
 66:                 if (($new = static::segment($index+1)) === null)
 67:                 {
 68:                     throw new \OutofBoundsException('Segment replace on "'.$url.'" failed. No segment exists for wildcard '.$wildcards.'.');
 69:                 }
 70:                 $segment = str_replace('*', $new, $segment);
 71:             }
 72:         }
 73: 
 74:         // re-assemble the path
 75:         $parts['path'] = '/'.implode('/', $segments);
 76: 
 77:         // and rebuild the url with the new path
 78:         if (empty($parts['host']))
 79:         {
 80:             // if a relative url was given, fake a host so we can remove it after building
 81:             $url = substr(http_build_url('http://__removethis__/', $parts), 22);
 82:         }
 83:         else
 84:         {
 85:             // a hostname was present, just rebuild it
 86:             $url = http_build_url('', $parts);
 87:         }
 88: 
 89:         // return the newly constructed url
 90:         return $url;
 91:     }

123:     /**
124:      * Creates a url with the given uri, including the base url
125:      *
126:      * @param   string  $uri            The uri to create the URL for
127:      * @param   array   $variables      Some variables for the URL
128:      * @param   array   $get_variables  Any GET urls to append via a query string
129:      * @param   bool    $secure         If false, force http. If true, force https
130:      * @return  string
131:      */
132:     public static function create($uri = null, $variables = array(), $get_variables = array(), $secure = null)
133:     {
134:         $url = '';
135:         $uri = $uri ?: static::string();
136: 
137:         // If the given uri is not a full URL
138:         if( ! preg_match("#^(http|https|ftp)://#i", $uri))
139:         {
140:             $url .= \Config::get('base_url');
141: 
142:             if ($index_file = \Config::get('index_file'))
143:             {
144:                 $url .= $index_file.'/';
145:             }
146:         }
147:         $url .= ltrim($uri, '/');
148: 
149:         // Add a url_suffix if defined and the url doesn't already have one
150:         if (substr($url, -1) != '/' and (($suffix = strrchr($url, '.')) === false or strlen($suffix) > 5))
151:         {
152:             \Config::get('url_suffix') and $url .= \Config::get('url_suffix');
153:         }
154: 
155:         if ( ! empty($get_variables))
156:         {
157:             $char = strpos($url, '?') === false ? '?' : '&';
158:             if (is_string($get_variables))
159:             {
160:                 $url .= $char.str_replace('%3A', ':', $get_variables);
161:             }
162:             else
163:             {
164:                 $url .= $char.str_replace('%3A', ':', http_build_query($get_variables));
165:             }
166:         }
167: 
168:         array_walk(
169:             $variables,
170:             function ($val, $key) use (&$url)
171:             {
172:                 $url = str_replace(':'.$key, $val, $url);
173:             }
174:         );
175: 
176:         is_bool($secure) and $url = http_build_url($url, array('scheme' => $secure ? 'https' : 'http'));
177: 
178:         return $url;
179:     }
