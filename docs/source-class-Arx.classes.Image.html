769:     // Courtest of: http://www.php.net/manual/en/function.imagecopymerge.php#88456
770:     private function imagecopymerge_alpha($dst_im, $src_im, $dst_x, $dst_y, $src_x, $src_y, $src_w, $src_h, $pct)
771:     {
772:         $pct /= 100;
773:         // Get image width and height
774:         $w = imagesx($src_im);
775:         $h = imagesy($src_im);
776:         // Turn alpha blending off
777:         imagealphablending($src_im, false);
778:         // Find the most opaque pixel in the image (the one with the smallest alpha value)
779:         $minalpha = 127;
780:         for ($x = 0; $x < $w; $x++) {
781:             for ($y = 0; $y < $h; $y++) {
782:                 $alpha = (imagecolorat($src_im, $x, $y) >> 24) & 0xFF;
783:                 if ($alpha < $minalpha) {
784:                     $minalpha = $alpha;
785:                 }
786:             }
787:         }
788:         // Loop through image pixels and modify alpha for each
789:         for ($x = 0; $x < $w; $x++) {
790:             for ($y = 0; $y < $h; $y++) {
791:                 // Get current alpha value (represents the TANSPARENCY!)
792:                 $colorxy = imagecolorat($src_im, $x, $y);
793:                 $alpha = ($colorxy >> 24) & 0xFF;
794:                 // Calculate new alpha
795:                 if ($minalpha !== 127) {
796:                     $alpha = 127 + 127 * $pct * ($alpha - 127) / (127 - $minalpha);
797:                 } else {
798:                     $alpha += 127 * $pct;
799:                 }
800:                 // Get the color index with new alpha
801:                 $alphacolorxy = imagecolorallocatealpha($src_im, ($colorxy >> 16) & 0xFF, ($colorxy >> 8) & 0xFF, $colorxy & 0xFF, $alpha);
802:                 // Set pixel with the new color + opacity
803:                 if (!imagesetpixel($src_im, $x, $y, $alphacolorxy)) return false;
804:             }
805:         }
806:         imagecopy($dst_im, $src_im, $dst_x, $dst_y, $src_x, $src_y, $src_w, $src_h);
