178:     {
179:         $sql = 'ALTER TABLE '.\DB::quote_identifier(\DB::table_prefix($table, $db ?: static::$connection), $db ?: static::$connection).' ';

298:     protected static function process_fields($fields, $prefix = '', $db = null)
299:     {
300:         $sql_fields = array();
301: 
302:         foreach ($fields as $field => $attr)
303:         {
304:             $attr = array_change_key_case($attr, CASE_UPPER);
305:             $_prefix = $prefix;
306:             if(array_key_exists('NAME', $attr) and $field !== $attr['NAME'] and $_prefix === 'MODIFY ')
307:             {
308:                 $_prefix = 'CHANGE ';
309:             }
310:             $sql = "\n\t".$_prefix;
311:             $sql .= \DB::quote_identifier($field);
312:             $sql .= (array_key_exists('NAME', $attr) and $attr['NAME'] !== $field) ? ' '.\DB::quote_identifier($attr['NAME'], $db ? $db : static::$connection).' ' : '';
313:             $sql .= array_key_exists('TYPE', $attr) ? ' '.$attr['TYPE'] : '';
314: 
315:             if(array_key_exists('CONSTRAINT',$attr))
316:             {
317:                 if(is_array($attr['CONSTRAINT']))
318:                 {
319:                     $sql .= "(";
320:                     foreach($attr['CONSTRAINT'] as $constraint)
321:                     {
322:                         $sql .= (is_string($constraint) ? "'".$constraint."'" : $constraint).", ";
323:                     }
324:                     $sql = rtrim($sql,', '). ")";
325:                 }
326:                 else
327:                 {
328:                     $sql .= '('.$attr['CONSTRAINT'].')';
329:                 }
330:             }
331: 
332:             $sql .= array_key_exists('CHARSET', $attr) ? static::process_charset($attr['CHARSET'], $db ? $db : static::$connection) : '';
333: 
334:             if (array_key_exists('UNSIGNED', $attr) and $attr['UNSIGNED'] === true)
335:             {
336:                 $sql .= ' UNSIGNED';
337:             }
338: 
339:             if(array_key_exists('DEFAULT', $attr))
340:             {
341:                 $sql .= ' DEFAULT '.(($attr['DEFAULT'] instanceof \Database_Expression) ? $attr['DEFAULT']  : \DB::escape($attr['DEFAULT']));
342:             }
343: 
344:             if(array_key_exists('NULL', $attr) and $attr['NULL'] === true)
345:             {
346:                 $sql .= ' NULL';
347:             }
348:             else
349:             {
350:                 $sql .= ' NOT NULL';
351:             }
352: 
353:             if (array_key_exists('AUTO_INCREMENT', $attr) and $attr['AUTO_INCREMENT'] === true)
354:             {
355:                 $sql .= ' AUTO_INCREMENT';
356:             }
357: 
358:             if (array_key_exists('PRIMARY_KEY', $attr) and $attr['PRIMARY_KEY'] === true)
359:             {
360:                 $sql .= ' PRIMARY KEY';
361:             }
362: 
363:             if (array_key_exists('FIRST', $attr) and $attr['FIRST'] === true)
364:             {
365:                 $sql .= ' FIRST';
366:             }
367:             elseif (array_key_exists('AFTER', $attr) and strval($attr['AFTER']))
368:             {
369:                 $sql .= ' AFTER '.\DB::quote_identifier($attr['AFTER'], $db ? $db : static::$connection);
370:             }
371: 
372:             if (array_key_exists('COMMENT', $attr))
373:             {
374:                 $sql .= ' COMMENT '.\DB::escape($attr['COMMENT'], $db ? $db : static::$connection);
375:             }
376: 
377:             $sql_fields[] = $sql;
378:         }
379: 
380:         return \implode(',', $sql_fields);
381:     }

455:      * @return   string    the formated foreign key string
456:      */
457:     public static function process_foreign_keys($foreign_keys, $db = null)
458:     {
459:         if ( ! is_array($foreign_keys))
460:         {
461:             throw new \Database_Exception('Foreign keys on create_table() must be specified as an array');
462:         }
463: 
464:         $fk_list = array();
465: 
466:         foreach($foreign_keys as $definition)
467:         {
468:             // some sanity checks
469:             if (empty($definition['key']))
470:             {
471:                 throw new \Database_Exception('Foreign keys on create_table() must specify a foreign key name');
472:             }
473:             if ( empty($definition['reference']))
474:             {
475:                 throw new \Database_Exception('Foreign keys on create_table() must specify a foreign key reference');
476:             }
477:             if (empty($definition['reference']['table']) or empty($definition['reference']['column']))
478:             {
479:                 throw new \Database_Exception('Foreign keys on create_table() must specify a reference table and column name');
480:             }
481: 
482:             $sql = '';
483:             ! empty($definition['constraint']) and $sql .= " CONSTRAINT ".\DB::quote_identifier($definition['constraint']);
484:             $sql .= " FOREIGN KEY (".\DB::quote_identifier($definition['key']).')';
485:             $sql .= " REFERENCES ".\DB::quote_identifier(\DB::table_prefix($definition['reference']['table'])).' (';
486:             if (is_array($definition['reference']['column']))
487:             {
488:                 $sql .= implode(', ', \DB::quote_identifier($definition['reference']['column']));
489:             }
490:             else
491:             {
492:                 $sql .= \DB::quote_identifier($definition['reference']['column']);
493:             }
494:             $sql .= ')';
495:             ! empty($definition['on_update']) and $sql .= " ON UPDATE ".$definition['on_update'];
496:             ! empty($definition['on_delete']) and $sql .= " ON DELETE ".$definition['on_delete'];
497: 
498:             $fk_list[] = "\n\t".ltrim($sql);
499:         }

501:         return ', '.implode(',', $fk_list);
