166:     public function getReadableFileSize($size, $retstring = null) {
167:         // adapted from code at http://aidanlister.com/repos/v/function.size_readable.php
168:         $sizes = array('bytes', 'kB', 'MB', 'GB', 'TB', 'PB', 'EB', 'ZB', 'YB');
169: 
170:         if ($retstring === null) { $retstring = '%01.2f %s'; }
171: 
172:         $lastsizestring = end($sizes);
173: 
174:         foreach ($sizes as $sizestring) {
175:             if ($size < 1024) { break; }
176:             if ($sizestring != $lastsizestring) { $size /= 1024; }
177:         }
178:         if ($sizestring == $sizes[0]) { $retstring = '%01d %s'; } // Bytes aren't normally fractional
179:         return sprintf($retstring, $size, $sizestring);
180:     }

182:     public function getReadableTime($time) {
183:         $ret = $time;
184:         $formatter = 0;
185:         $formats = array('ms', 's', 'm');
186:         if($time >= 1000 && $time < 60000) {
187:             $formatter = 1;
188:             $ret = ($time / 1000);
189:         }
190:         if($time >= 60000) {
191:             $formatter = 2;
192:             $ret = ($time / 1000) / 60;
193:         }
194:         $ret = number_format($ret,3,'.','') . ' ' . $formats[$formatter];
195:         return $ret;
