 81:     /**
 82:      * Parses HTML and extracts Open Graph data, this assumes
 83:      * the document is at least well formed.
 84:      *
 85:      * @param $HTML    HTML to parse
 86:      * @return OpenGraph
 87:      */
 88:     static private function _parse($HTML) {
 89:         $old_libxml_error = libxml_use_internal_errors(true);
 90: 
 91:         $doc = new DOMDocument();
 92:         $doc->loadHTML($HTML);
 93: 
 94:         libxml_use_internal_errors($old_libxml_error);
 95: 
 96:         $tags = $doc->getElementsByTagName('meta');
 97:         if (!$tags || $tags->length === 0) {
 98:             return false;
 99:         }
100: 
101:         $page = new self();
102: 
103:         $nonOgDescription = null;
104: 
105:         foreach ($tags AS $tag) {
106:             if ($tag->hasAttribute('property') &&
107:                 strpos($tag->getAttribute('property'), 'og:') === 0) {
108:                 $key = strtr(substr($tag->getAttribute('property'), 3), '-', '_');
109:                 $page->_values[$key] = $tag->getAttribute('content');
110:             }
111: 
112:             //Added this if loop to retrieve description values from sites like the New York Times who have malformed it.
113:             if ($tag ->hasAttribute('value') && $tag->hasAttribute('property') &&
114:                 strpos($tag->getAttribute('property'), 'og:') === 0) {
115:                 $key = strtr(substr($tag->getAttribute('property'), 3), '-', '_');
116:                 $page->_values[$key] = $tag->getAttribute('value');
117:             }
118:             //Based on modifications at https://github.com/bashofmann/opengraph/blob/master/src/OpenGraph/OpenGraph.php
119:             if ($tag->hasAttribute('name') && $tag->getAttribute('name') === 'description') {
120:                 $nonOgDescription = $tag->getAttribute('content');
121:             }
122: 
123:         }
124:         //Based on modifications at https://github.com/bashofmann/opengraph/blob/master/src/OpenGraph/OpenGraph.php
125:         if (!isset($page->_values['title'])) {
126:             $titles = $doc->getElementsByTagName('title');
127:             if ($titles->length > 0) {
128:                 $page->_values['title'] = $titles->item(0)->textContent;
129:             }
130:         }
131:         if (!isset($page->_values['description']) && $nonOgDescription) {
132:             $page->_values['description'] = $nonOgDescription;
133:         }
134: 
135:         //Fallback to use image_src if ogp::image isn't set.
136:         if (!isset($page->values['image'])) {
137:             $domxpath = new DOMXPath($doc);
138:             $elements = $domxpath->query("//link[@rel='image_src']");
139: 
140:             if ($elements->length > 0) {

142:                 if ($domattr) {

144:                     $page->_values['image_src'] = $domattr->value;
145:                 }
146:             }
147:         }
148: 
149:         if (empty($page->_values)) { return false; }
